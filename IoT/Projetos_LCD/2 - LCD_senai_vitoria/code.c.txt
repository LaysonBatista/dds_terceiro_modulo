#include <LiquidCrystal.h>
LiquidCrystal lcd(12,11,10,5,4,3,2);

int backLight = 13;

void setup()
{
  pinMode(backLight, OUTPUT);
  digitalWrite(backLight, HIGH);
}

void loop()
{ 
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SENAI");
  delay(1000);
  lcd.clear();
  
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("VITORIA");
  delay(1000);
  lcd.clear();
  
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("DES. DE");
  delay(1000);
  lcd.clear();
  
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SISTEMAS");
  delay(1000);
  lcd.clear();
}
